@AttributeOverrides(value = {
                          @AttributeOverride(name = "areaCode", column = @Column(name = "AREA_CODE")),
                          @AttributeOverride(name = "phNumber", column = @Column(name = "PH_NUMBER"))
                          })

@OneToOne
@JoinColumns({
			@JoinColumn(name = "areaCode"),
			@JoinColumn(name = "phNumber")
})
